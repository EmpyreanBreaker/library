function deleteBook(bookIndex) {
    for (let i = 0; i < myLibrary.length; i++) {
        if (bookIndex === myLibrary[i].id) {
            myLibrary.splice(i, 1);
            break;
        }
    }
    displayBooks(myLibrary);
}
